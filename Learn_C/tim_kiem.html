Tìm kiếm

Linear Search (Tìm kiếm tuyến tính)

Độ phức tập: O(N)

Tư tưởng: duyệt từ đầu đến cuối. 

 

        int LinearSearch(int a[], int n, int x)
        {
            for(int i=0;i < n;i++)
            {
                if(a[i] == x) return 1;
            }
            return 0;
        }

Binary Search (Tìm kiếm nhị phân)

Độ phức tập: O(LogN)

Điều kiện để tìm kiếm là mảng đã được sắp xếp rồi

Tư tưởng: Tìm kiếm trong đoạn từ [left, right] của mảng, ở mỗi bước thuật toán tìm vị trị mid ở giữa đoạn left, right. Nếu phần tử cần tìm kiếm bằng phần tử ở vị trí mid thì kết luận là tìm thấy, nếu không ta có thể giảm 1 nửa đoạn tìm kiếm xuống và tiếp tục tìm kiếm ở bên trái hay bên phải của mid

 

                  #include < stdio.h >
                    int LinearSearch(int a[], int n, int x)
                    {
                        int l = 0, r = n -1;
                        while(l <= r)
                        {
                            int m = (l + r) / 2;
                            if(a[m] == x) return 1;
                            else {
                                if(a[m] < x) l = m + 1;
                                else r = m - 1;
                            }
                        }
                        return 0;
                    }
                    int main()
                    {
                        int n,x; scanf("%d%d",&n,&x);
                        int a[n];
                        for(int i=0;i < n;i++) scanf("%d",&a[i]);
                        printf("%d\n", LinearSearch(a,n,x));
                    }

Các biến thể của tìm kiếm nhị phân

1) Vị trí đầu tiên trong mảng tăng dần

Độ phức tạp: O(LogN)

 

                  #include < stdio.h >
                    // Tìm chỉ số đầu tiên của phần tử x trong mảng a đã được sắp xếp, nếu ko tìm thấy trả về -1
                    int First_linear_search(int a[], int n, int x)
                    {
                         int res = -1, l = 0, r = n - 1;
                         while(l <= r)
                         {
                            int m = (l + r) / 2;
                            if(a[m] == x)
                            {
                                res = m;
                                // tìm thêm ở bên trai xem có còn x ko?
                                r = m - 1;
                            }
                            else if(a[m] < x) l = m + 1;
                            else r = m - 1;
                         }
                         return res;
                    }
                    int main()
                    {
                        int n,x; scanf("%d%d",&n,&x);
                        int a[n];
                        for(int i=0;i < n;i++) scanf("%d",&a[i]);
                        printf("%d\n", First_linear_search(a,n,x));
                    }

3) Vị trí đầu tiên lớn hơn hoặc bằng X trong mảng tăng dần

Độ phức tạp: O(LogN)

 

                  #include < stdio.h >
                    // Tìm chỉ đầu tiên lớn hơn hoặc bằng phần tử x trong mảng a đã được sắp xếp, nếu ko tìm thấy trả về -1
                    int First_linear_search(int a[], int n, int x)
                    {
                         int res = -1, l = 0, r = n - 1;
                         while(l <= r)
                         {
                            int m = (l + r) / 2;
                            if(a[m] >= x)
                            {
                                res = m;
                                // tìm thêm ở bên trái
                                r = m - 1;
                            }
                            else   l = m + 1;   
                         }
                         return res;
                    }
                    int main()
                    {
                        int n,x; scanf("%d%d",&n,&x);
                        int a[n];
                        for(int i=0;i < n;i++) scanf("%d",&a[i]);
                        printf("%d\n", First_linear_search(a,n,x));
                    }
