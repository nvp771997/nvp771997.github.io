S(n) =

  • Bài toán con nhỏ nhất: nếu n < 10 thì trả về n
  • Công thức truy hồi: n%10 + S(n/10), n>=10



         int tong(int n){
         if(n < 10) return n;
         else return n%10 + tong(n/10);
         }
Đếm số lượng chữ số của n
  • Bài toán con nhỏ nhất: nếu n < 10 thì trả về 1
  • Công thức truy hồi: 1 + C(n/10)


 
           int DCS(int n){
                if(n < 10) return 1;
                else return 1 + DCS(n/10);
           }

Tìm chữ số đầu tiên của n

  • Bài toán con nhỏ nhất: nếu n < 10 thì trả về n
  • Công thức truy hồi: SDT(n/10)


      int SDT(int n){
                if(n < 10) return n;
                else return SDT(n/10);
           }

                      int chan(long long n){
                if(n < 10) 
                    if(n%2==0) return n;
                    else return 0;
                else{
                    if(n % 10 % 2 == 0)
                        return n%10 + chan(n/10);
                    else
                        return chan(n/10);
                } 
                    
           }
           int le(long long n){
                if(n < 10) 
                    if(n%2==0) return 0;
                    else return n;
                else{
                    if(n % 10 % 2 == 1)
                        return n%10 + le(n/10);
                    else
                        return le(n/10);
                } 
                    
           }
