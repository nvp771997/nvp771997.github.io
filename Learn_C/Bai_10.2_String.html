Đếm tần suất của các ký tự trong xâu



      //In ra theo thứ tự bảng mã ascii
      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
      #include < stdlib.h >
        int main()
        { 
          char c[700];
          gets(c);
          int cnt[256] = {0};
          for(int i=0;i < strlen(c);i++)
          {
            cnt[c[i]]++; //Đếm tần suất bằng số của bản mã ascii
          }
          for(int i=0;i < 256;i++)
          {
            if(cnt[i]) printf("%c %d\n",i,cnt[i]);
          }
        }
    


      //In ra theo thứ tự xuất hiện đầu tiên và ko được trùng (tìm ký tự xuất hiện nhiều nhất hoặc ít nhất)
      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
      #include < stdlib.h >
        int main()
        { 
          char c[700];
          gets(c);
          int cnt[256] = {0};
          for(int i=0;i < strlen(c);i++)
          {
            cnt[c[i]]++; //Đếm tần suất bằng số của bản mã ascii
          }
          for(int i=0;i < strlen(c);i++)
          {
             if(cnt[c[i]])
             {
                printf("%c %d\n",c[i],cnt[c[i]]);
                cnt[c[i]] = 0;
             }
          }
        }
