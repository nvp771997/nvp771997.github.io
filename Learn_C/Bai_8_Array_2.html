Cộng trừ 2 ma trận

Điều kiện để cộng hoặc trừ 2 ma trận là số hàng và số cột của 2 ma trận phải giống nhau



      #include < stdio.h >
        void nhap(int a[][100], int n, int m)
        {
            for(int i=0;i < n;i++)
            {
                for(int j=0;j < m;j++) scanf("%d",&a[i][j]);
            }
        }
        void in(int c[][100], int n, int m)
        {
            for(int i=0;i < n;i++)
            {
                for(int j=0;j < m;j++) printf("%d ",c[i][j]);
                    printf("\n");
            }
        }
        int main()
        {
            int a[100][100], b[100][100], c[100][100];
            int n,m; scanf("%d%d",&n,&m);
            nhap(a,n,m);
            nhap(b,n,m);
            //Cộng 2 ma trận
            for(int i=0;i < n;i++)
            {
                for(int j=0;j < m;j++) c[i][j] = a[i][j] + b[i][j];
            }
            in(c,n,m);
        }

Nhân 2 ma trận

Công thức: a[n][m] x b[m][p] = c[n][q]



      #include < stdio.h >
        int main()
        {
            int n,m,p; scanf("%d%d%d",&n,&m,&p);
            int a[n][m], b[m][p], c[n][p];
            // Nhập ma trận a[n][m]
            for(int i=0;i < n;i++){
                for(int j=0;j < m;j++) scanf("%d",&a[i][j]);
            }
            // Nhập ma trận b[m][p]
            for(int i=0;i < m;i++){
                for(int j=0;j < p;j++) scanf("%d",&b[i][j]);
            }
            // Nhân 2 ma trận
            for(int i=0;i < n;i++)
            {
                for(int j=0;j < p;j++)
                {
                    c[i][j]=0;
                    for(int k=0;k < m;k++)
                    {
                        c[i][j] += a[i][k] * b[k][j];
                    }
                }
            }
            // In ma trận
            for(int i=0;i < n;i++){
                for(int j=0;j < p;j++) printf("%d ",c[i][j]);
                    printf("\n");
            }
        }
