Các thao tác cơ bản trên mảng Struct

Các hàm sắp xếp trong Struct

- Kiểu cấu trúc (Structure) là kiểu dữ liệu bao gồm nhiều thành phần có kiểu khác nhau, mỗi thành phần được gọi là một trường (field)

- Cho phép you tự định nghĩa kiểu dữ liệu mới

Cú pháp

 

                struct struct_name{
                      // Danh sách các thuộc tính
                };

Để truy cập vào các trường dữ liệu của struct ta dùng toán tử '.'

Mảng Struct

      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
      #include < stdlib.h >
        // Khai báo struct
        struct sinhvien{
             char ten[70];
            char lop[70];
            double gpa;
        };
        typedef struct sinhvien sinhvien; //Định nghĩa tên mới của struct sinhvien
        int main()
        { 
           int n; 
           printf("Nhap so luong sv: "); scanf("%d",&n);
           sinhvien a[n];
           for(int i=0;i < n;i++)
           {
             getchar(); //Xóa phím enter
             printf("Nhap ten: "); gets(a[i].ten);
             printf("Nhap lop: "); gets(a[i].lop);
             printf("Nhap gpa: "); scanf("%lf",&a[i].gpa); //Chôi lệnh nếu ko dùng getchar()
           }
           printf("Danh sach sinh vn:\n");
           for(int i=0;i < n;i++){
            printf("%s %s %.2lf\n",a[i].ten, a[i].lop, a[i].gpa);
           }
        }

Hàm nhập thông tin và xuất thông tin SV



      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
      #include < stdlib.h >
        // Khai báo struct
        struct sinhvien{
             char ten[70];
            char lop[70];
            double gpa;
        };
        typedef struct sinhvien sinhvien; //Định nghĩa tên mới của struct sinhvien
        //Hàm nhập thông tin
        sinhvien nhap(){
            sinhvien x; //Biến tạm
            printf("Nhap ten: "); gets(x.ten);
            printf("Nhap lop: "); gets(x.lop);
            printf("Nhap gpa: "); scanf("%lf",&x.gpa);
            return x;
        }
        //Hàm in thông tin
        void in(sinhvien x){
            printf("%s %s %.2lf\n",x.ten, x.lop, x.gpa);
        }
        int main()
        { 
           int n; 
           printf("Nhap so luong sv: "); scanf("%d",&n);
           sinhvien a[n];
        
           for(int i=0;i < n;i++)
           {
             getchar(); // Xóa phím enter sau hàm scanf
             a[i] = nhap(); //Phải gán lại cho a[i]
           }
           printf("Danh sach sinh vn:\n");
           for(int i=0;i < n;i++){
             in(a[i]);
           }
        }

Con trỏ kiểu Struct

Tương tự như kiểu dữ liệu thông thường, bạn có thể khai báo con trỏ struct. Nhưng khi truy cập vào các trường dữ liệu của biến con trỏ struct ta dùng -> thay vì dùng toán tử '.' 



      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
      #include < stdlib.h >
        struct sinhvien{
            char ten[70];
            char lop[70];
            double gpa;
        };
        typedef struct sinhvien sinhvien;
        int main()
        {
            sinhvien x;
            strcpy(x.ten, "The Assassin Naktoh");
            strcpy(x.lop, "Class07");
            x.gpa = 3.7;
            sinhvien *ptr = &x;
            //Có thể truy cập trường dữ liệu của biến con trỏ bằng dấu ->
            printf("%s %s %.2lf\n",ptr->ten, ptr->lop, ptr->gpa);
            printf("%s %s %.2lf\n",(*ptr).ten, (*ptr).lop, (*ptr).gpa); //Có thể giải tham chiếu con trỏ ptr và truy cập bằng '.'
        }

Hàm nhập thông tin và xuất thông tin SV bằng cách truyền tham chiếu bằng con trỏ

       
      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
      #include < stdlib.h >
        // Khai báo struct
        struct sinhvien{
             char ten[70];
            char lop[70];
            double gpa;
        };
        typedef struct sinhvien sinhvien; //Định nghĩa tên mới của struct sinhvien
        //Hàm nhập thông tin
        void nhap(sinhvien *x){
            printf("Nhap ten: "); gets(x->ten);
            printf("Nhap lop: "); gets(x->lop);
            printf("Nhap gpa: "); scanf("%lf",&x->gpa);
        }
        //Hàm in thông tin
        void in(sinhvien x){
            printf("%s %s %.2lf\n",x.ten, x.lop, x.gpa);
        }
        int main()
        { 
           int n; 
           printf("Nhap so luong sv: "); scanf("%d",&n);
           sinhvien a[n];
        
           for(int i=0;i < n;i++)
           {
             getchar(); // Xóa phím enter sau hàm scanf
             nhap(&a[i]); //Truyền địa chỉ của a[i]
           }
           printf("Danh sach sinh vn:\n");
           for(int i=0;i < n;i++){
             in(a[i]);
           }
        }
