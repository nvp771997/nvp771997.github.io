String

Các hàm xử lý xâu thông dụng

Các bài toán liên quan đến tầng suất trong String

- Chuỗi ký tự là 1 mảng ký tự và kết thúc bằng ký tự '\0' (ký tự NULL)

- Để sữ dụng được String thì phải theo 2 thư viện string.h và ctype.h

Khai báo String



       #include < stdio.h >
       int main()
       {
       char c[] = "Nakroth"; //Xâu ký tự thì đặt trong dấu nháy ""
       char p[] = {'N', 'a', 'k', 'r', 'o', 't', 'h', '\0'};
       
       printf("%s",c);
       printf("\n");
       for(int i=0;i < 7;i++) printf("%c",p[i]);  
       printf("\n");
       printf("%s",p);
       }

Nhập String

Nhập xâu không có dấu cách



         #include < stdio.h >
         int main()
         {
         char c[70];
         scanf("%s",c); //Không nhập được dấu cách
         printf("%s\n",c);

         }

Nhập xâu có dấu cách bằng gets

 

        #include < stdio.h >
        #include < string.h >
        #include < ctype.h >
        int main()
        {
            char c[70];
            gets(c); //Nhập được cả dấu cách
            printf("%s\n",c);
        }

Nhập xâu bằng fgets mới và an toàn hơn



      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
      int main()
      {
         char c[70];
         fgets(c,70,stdin); //Kiểu mới và an toàn hơn gets
         printf("%s\n",c);
      }

Chôi lệnh

Lỗi chôi lệnh khi dùng scanf trước gets

 

       #include < stdio.h >
       #include < string.h >
       #include < ctype.h >
       int main()
       {
         int n; scanf("%d",&n);
         char c[70];
         gets(c);
         printf("%s %d\n",c,n);
       }

Cách khắc phục là xóa bộ nhớ đệm bằng cách fflush(stdin) nhưng mà cách này sẽ xóa toàn bộ bộ nhớ đệm. Nên chúng ta chỉ cần dụng hàm getchar(); đọc ký tự từ bộ đệm bàn phím là được.



       #include < stdio.h >
       #include < string.h >
       #include < ctype.h >
       int main()
       {
           int n; scanf("%d",&n);
           char c[70];
           //getchar(); //Hàm đọc 1 ký tự từ bàn phím, được sữ dụng khi có scanf trước gets
           while(getchar() != '\n'); //Tránh trường hợp nhập nhiều lần dấu cách
           gets(c);
           printf("%s %d\n",c,n);
       }

Các hàm thường gặp trong String

strlen

strlen: trả về số lượng có trong chuỗi



          #include < stdio.h >
          #include < string.h >
          #include < ctype.h >
          int main()
          {
              char c[70];
              gets(c);
              printf("So luong ky tu: %d\n", strlen(c));
          }

Trong hàm strlen thì sự khác nhau giữa gets và fgets: fgets sẽ đọc cả kí tự enter và cho vào xâu, ngược lại gets thì không

Cách loại bỏ kí tự enter nếu bạn dùng fgets đó là cho kí tự đó thành kí tự null.



      #include < stdio.h >
        #include < string.h >
        #include < ctype.h >
        int main()
        {
            char c[70], d[70];
            gets(c);
            fgets(d, 70, stdin);
            printf("%d %d\n", strlen(c), strlen(d));
            d[strlen(d)-1] = '\0';
            printf("%d %d\n",strlen(c), strlen(d));
        }
