Các hàm xử lý xâu thông dụng

Các hàm này nằm trong thư viện ctype.h

Các hàm kiểm tra ký tự

  • isdigit(char c) : Kiểm tra chữ số
  • islower(char c) : Kiểm tra chữ in thường
  • isupper(char c) : Kiểm tra in hoa
  • isalpha(char c) : Kiểm tra chữ cái
  • tolower(char c) : Chuyển thành chữ in thường (ko làm thay đổi giá trị thực của chuỗi)
  • tolower(char c) : Chuyển thành chữ in thường (ko làm thay đổi giá trị thực của chuỗi)
  • toupper(char c) : Chuyển thành chữ in hoa 


       #include < stdio.h >
       #include < string.h >
       #include < ctype.h >
       int main()
       {  
           char c[] = "Nakroth777!!!!OK";
           int lower = 0, upper = 0, digit = 0, alpha = 0;
           for(int i=0;i < strlen(c);i++)
           {
               if(isdigit(c[i])) ++digit; //Kiểm tra chữ số
               if(islower(c[i])) ++lower; //Kiểm tra chữ in thường
               if(isupper(c[i])) ++upper; //Kiểm tra chữ in hoa
               if(isalpha(c[i])) ++alpha; //Kiểm tra chữ cái
           }
           printf("%d %d %d %d\n",digit, lower, upper, alpha);
       }

Hàm strlwr

Cú pháp: char* strlwr(char[]);

Chức năng: Chuyển xâu kí tự về dạng in thường, ngoài ra còn trả về con trỏ tới mảng này sau khi chuyển đổi

 

      #include < stdio.h >
      #include < string.h >
        int main()
        {
            char c[] = "NAKROTH";
            char *a = strlwr(c); // Hàm chuyển chuổi chữ in thành chữ thường
            printf("%s\n",c);
            printf("%s\n",a);  
            //Output: nakroth  
        }

Xây dụng hàm strlwr



      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
        void lower(char c[]){
            for(int i=0;i < strlen(c);i++)
            {
                c[i] = tolower(c[i]); // Phải gán giá trị cho c[i] vì tolwer ko làm thay đổi mảng ký tự
            }
        }
        int main()
        {
            char c[] = "NAKROTH";
            lower(c);

Hàm strcmp

Cú pháp: int strcmp(char a[], char b[]);

Chức năng: Hàm này dùng để so sánh 2 xâu theo thứ tự từ điển

Giá trị trả về:
  • Nếu a < b trả về -1
  • Nếu a > b trả về 1
  • Nếu a = b trả về 0


      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
         
        int main()
        {
           char c[] = "Nakroth";
           char d[] = "ZNakroth";
           char p[] = "Nakroth";
           printf("%d ", strcmp(c,d));
           printf("%d ", strcmp(c,p));
           printf("%d ", strcmp(d,p));
           // Output: -1 0 1
        }

Hàm strcat

Cú pháp: char* strcat(char dich[], char nguon[]);

Chức năng: Nối xâu nguồn vào xâu đích, ngoài ra còn trả về con trỏ tới mảng này sau khi nối.



      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
         
        int main()
        {
           char c[] = "The Assassin ";
           char d[] = "Nakroth";
           strcat(c,d); // Nối chuỗi d vào chuổi c
           printf("%s\n",c);
           //Output: The Assassin Nakroth
        }

Hàm strcpy

Cú pháp: char* strcpy(char dich[], char nguon[]);

Chức năng: Copy nội dung từ xâu nguồn vào xâu đích, chú ý rằng trước khi copy, toàn bộ nội dung của xâu đích sẽ bị xóa hết. Ngoài ra còn trả về con trỏ tới mảng này sau khi copy.



      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
         
        int main()
        {
           char c[] = "The Assassin ";
           char d[] = "Nakroth";
           strcpy(c,d); // Trước khi copy thì nội dung của chuỗi c sẽ bị xóa hết.
           printf("%s\n",c);
           //Output: Nakroth
        }

Hàm strstr

Cú pháp: char* strstr(char c[], char d[]);

Chức năng: Trả về con trỏ tới vị trí xuất hiện đầu tiên của xâu d trong xâu c, nếu xâu d không xuất hiện trong xâu c nó trả về con trỏ NULL. Hàm này được dùng để kiểm tra sự tồn tại của xâu con.

      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
         
        int main()
        {
           char c[] = "Nakroth123OK";
           char d[] = "th";
           if(strstr(c,d) != NULL) printf("Found\n");
           else printf("Not Found\n");
           //Output: Fuond
        }

Hàm memset

Chức năng: Hàm memset được dùng để gán giá trị cho các ô nhớ, các bạn có thể dùng hàm này để gán giá trị 0 hoặc -1 cho mảng int.



      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
         
        int main()
        {
           int a[7],b[7];
           memset(a, 0, sizeof(a));
           for(int i=0;i < 7;i++) printf("%d ",a[i]);
           printf("\n");
           memset(b, -1, sizeof(b));
           for(int i=0;i < 7;i++) printf("%d ",b[i]);
           //Output: 0 0 0 0 0 0 0 và -1 -1 -1 -1 -1 -1 -1
                   
        }

Hàm atoi, atoll

Chức năng: chuyển đổi các xâu kí tự sang số tương ứng.

  • atoi : chuyển xâu sang số int
  • atoll : chuyển xâu sang số long long


      #include < stdio.h >
      #include < string.h >
      #include < ctype.h >
      #include < stdlib.h > // Thêm thư viện này để sữ dụng atoi và atoll 
        int main()
        {
           char c[] = "007"; // Chỉ sữ dụng chữ số
           char d[] = "7000000000";
           int n = atoi(c);
           long long p = atoll(d); 
           printf("%d\n%lld\n",n,p);
           //Output: 7 7000000000
        }

Hàm strtok

Hàm strtok có chức năng tách các từ trong xâu theo một kí tự chỉ đinh cho trước, thông thường ta sẽ dùng strtok để tách các từ trong xâu theo dấu cách



          #include < stdio.h >
          #include < string.h >
          #include < ctype.h >
          #include < stdlib.h >
            int main()
            { 
              char c[] = "The Assassin Nakroth";
              char *p = strtok(c, " ");
              while(p != NULL)
              {
                printf("%s\n",p);
                p = strtok(NULL, " ");
              }     
            }
            Output:   The
                      Assassin
                      Nakroth

Hàm strtok còn có thể được dùng để tách theo nhiều kí tự khác nhau.
