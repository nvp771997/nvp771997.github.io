Các hàm đọc file

Đọc 1 kí tự từ file

Đọc 1 kí tự từ file, nếu tới cuối file và không đọc được nữa thì hàm này trả về EOF



        Syntax
        fgetc(file_pointer);
     


        VD:
       #include < stdio.h>
       #include < string.h>
       int main(){
          FILE *f;
          f = fopen("input.txt", "r"); //Read
           if(f == NULL){
              printf("Khong ton tai\n");
           }
           else{
               char c = fgetc(f);
               while(c != EOF){ //nếu tới cuối file và không đọc được nữa thì hàm này trả về EOF
                    printf("%c",c);
                    c = fgetc(f);
               }
           }
          fclose(f);
       }

Đọc từ bàn phím



        Syntax
        fscanf(file_pointer, format, variables);
     


        VD:
       #include < stdio.h>
       #include < string.h>
       int main(){
          FILE *f;
          f = fopen("input.txt", "r"); //Read
           if(f == NULL){
              printf("Khong ton tai\n");
           }
           else{
              //Đọc số
              int n; fscanf(f, "%d", &n);
              int a[n];
              for(int i=0;i < n;i++) fscanf(f, "%d",a+i);
              for(int i=0;i < n;i++) printf("%d ",a[i]);
              printf("\n");
              //Đọc chuỗi
              char c[70];
              fscanf(f, "%s",c);
              printf("%s\n",c);
           }
          fclose(f);
       }
     


        Đọc hết các số bằng fscanf
       #include < stdio.h>
       #include < string.h>
       int main(){
          FILE *f;
          f = fopen("input.txt", "r"); //Read
           if(f == NULL){
              printf("Khong ton tai\n");
           }
           else{
              //Đọc số
               int n;
               while(fscanf(f, "%d",&n) != -1){ //Khi đọc hết các giá trị thì trả về -1
                   printf("%d ",n);
                 }
               }
          fclose(f);
       }

Đọc cả 1 dòng trong file



        Syntax
        fgets(buffer, size, file_pointer);
     


        VD: Đọc cả 1 dòngh
       #include < stdio.h>
       #include < string.h>
       int main(){
          FILE *f;
          f = fopen("input.txt", "r"); //Read
           if(f == NULL){
              printf("Khong ton tai\n");
           }
           else{
               char c[100];
               fgets(c, 100, f); //Chôi lệnh nếu đọc nhiều hàng
               printf("%s\n",c);
           }
          fclose(f);
       }
