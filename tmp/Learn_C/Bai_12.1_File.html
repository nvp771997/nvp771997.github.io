Các hàm ghi file

Ghi 1 ký tự vào file


        Syntax
        fputc(char, file_pointer);
     


        VD:
       #include 
       #include 
       int main(){
          FILE *f;
          f = fopen("input.txt", "w"); //write
          char c[] = "Nakroth";
          for(int i=0;i < strlen(c);i++){
          fputc(c[i], f); //Ghi 1 ký tự vào file
          }
          fclose(f);
       }

Ghi 1 xâu vào file



        Syntax
        fputs(str, file_pointer);
     


        VD:
       #include < stdio.h>
       #include < string.h>
       int main(){
          FILE *f;
          f = fopen("input.txt", "w"); //write
          char c[] = "Nakroth\n";
          fputs(c,f); //Ghi 1 xâu tự vào file
          fclose(f);
       }

Ghi vào file thông qua con trỏ

Hàm được sữ dụng nhiều



        Syntax
        fprintf(file_pointer, format, variables);
     


        VD:
       #include < stdio.h>
       #include < string.h>
       int main(){
          FILE *f;
          f = fopen("input.txt", "w"); //write
          for(int i=0;i <= 70;i++){
                fprintf(f, "%d ", i);
          }
          fclose(f);
       }
